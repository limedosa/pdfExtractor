from selenium import webdriver from selenium.webdriver.chrome.options import Options import os # Set paths html_file = r"C:\Users\ldomi\OneDrive - WideNet AI\Documents\GitHub\pdfCreator\jefferies_docs\ef20044774_424b2.htm" pdf_file = html_file.replace(".htm", ".pdf") # Configure Chrome options = Options() options.add_argument('--headless') options.add_argument('--disable-gpu') options.add_argument(f'--print-to-pdf={pdf_file}') # Open Chrome and convert HTML to PDF driver = webdriver.Chrome(options=options) driver.get(f"file:///{html_file}") driver.quit() print(f"PDF saved at: {pdf_file}")
